```swift
import CoreData

@objc(Event)
public class Event: NSManagedObject {
    @NSManaged public var id: UUID
    @NSManaged public var type: String // "feed", "sleep", "diaper", "growth"
    @NSManaged public var timestamp: Date
    @NSManaged public var endDate: Date?
    @NSManaged public var notes: String?
    @NSManaged public var metadata: Data? // JSON for type-specific data
}

@objc(Baby)
public class Baby: NSManagedObject {
    @NSManaged public var id: UUID
    @NSManaged public var name: String
    @NSManaged public var birthDate: Date
    @NSManaged public var photoData: Data?
    @NSManaged public var events: NSSet?
}

enum FeedType: String, Codable {
    case breastLeft, breastRight, bottle, solid
}

struct FeedMetadata: Codable {
    let feedType: FeedType
    let duration: TimeInterval? // breast
    let volume: Double? // bottle (ml)
    let unit: String? // "ml" or "oz"
}

struct SleepMetadata: Codable {
    let quality: SleepQuality? // good, fair, poor
}

enum SleepQuality: String, Codable {
    case good, fair, poor
}

struct DiaperMetadata: Codable {
    let diaperType: DiaperType
}

enum DiaperType: String, Codable {
    case wet, dirty, mixed
}

struct GrowthMetadata: Codable {
    let weight: Double? // kg
    let height: Double? // cm
    let headCircumference: Double? // cm
    let unit: String // "metric" or "imperial"
```

### Live Activity后台计时器

```swift
import ActivityKit

struct SleepActivityAttributes: ActivityAttributes {
    public struct ContentState: Codable, Hashable {
        var startTime: Date
        var babyName: String
    }
    var babyName: String
}

final class LiveActivityManager {
    static let shared = LiveActivityManager()
    private var currentActivity: Activity<SleepActivityAttributes>?
    
    func startSleepTimer(babyName: String) {
        guard ActivityAuthorizationInfo().areActivitiesEnabled else { return }
        
        let attributes = SleepActivityAttributes(babyName: babyName)
        let state = SleepActivityAttributes.ContentState(
            startTime: Date(),
            babyName: babyName
        )
        
        do {
            currentActivity = try Activity.request(
                attributes: attributes,
                content: .init(state: state, staleDate: nil),
                pushType: nil
            )
        } catch {
            print("Live Activity failed: \(error)")
        }
    }
    
    func stopSleepTimer() {
        Task {
            await currentActivity?.end(
                .init(state: .init(startTime: Date(), babyName: ""), staleDate: nil),
                dismissalPolicy: .immediate
            )
            currentActivity = nil
        }
    }
    
    func updateSleepTimer() {
        guard let activity = currentActivity else { return }
        Task {
            await activity.update(.init(
                state: activity.content.state,
                staleDate: nil
            ))
        }
    }
}
```

### CloudKit同步引擎

```swift
import CloudKit

final class CloudKitSyncService: ObservableObject {
    private let container: CKContainer
    private let privateDB: CKDatabase
    private let zone: CKRecordZone
    private var changeToken: CKServerChangeToken?
    
    init() {
        self.container = CKContainer(identifier: "iCloud.com.lullab.app")
        self.privateDB = container.privateCloudDatabase
        self.zone = CKRecordZone(zoneName: "LullabZone")
    }
    
    func createCustomZone() async throws {
        try await privateDB.save(zone)
    }
    
    func pushEvent(_ event: Event) async throws {
        let record = CKRecord(recordType: "Event", zoneID: zone.zoneID)
        record["type"] = event.type
        record["timestamp"] = event.timestamp
        record["endDate"] = event.endDate
        record["notes"] = event.notes
        record["metadata"] = event.metadata
        
        try await privateDB.save(record)
    }
    
    func pullChanges() async throws {
        let configuration = CKFetchRecordZoneChangesOperation.ZoneConfiguration()
        configuration.previousServerChangeToken = changeToken
        
        let operation = CKFetchRecordZoneChangesOperation(
            zoneConfigurations: [zone.zoneID: configuration]
        )
        
        operation.recordChangedBlock = { record in
            self.processIncomingRecord(record)
        }
        
        operation.recordZoneChangeTokensUpdatedBlock = { zoneID, token, _ in
            self.changeToken = token
            self.saveChangeToken(token)
        }
        
        operation.fetchRecordZoneChangesResultBlock = { result in
            switch result {
            case .success:
                break
            case .failure(let error):
                print("Sync error: \(error)")
            }
        }
        
        privateDB.add(operation)
    }
    
    private func processIncomingRecord(_ record: CKRecord) {
        let context = PersistenceController.shared.container.newBackgroundContext()
        context.perform {
            let fetchRequest: NSFetchRequest<Event> = Event.fetchRequest()
            fetchRequest.predicate = NSPredicate(format: "ckRecordID == %@", record.recordID.recordName)
            
            if let existing = try? context.fetch(fetchRequest).first {
                self.updateEvent(existing, from: record, in: context)
            } else {
                self.createEvent(from: record, in: context)
            }
            
            try? context.save()
        }
    }
    
    private func saveChangeToken(_ token: CKServerChangeToken?) {
        guard let data = try? NSKeyedArchiver.archivedData(withRootObject: token as Any, requiringSecureCoding: false) else { return }
        UserDefaults.standard.set(data, forKey: "cloudkit_change_token")
    }
}
```

## 🎨 UI设计规范

### 设计哲学: Dark-First + One-Handed + Calm

**三大设计原则**:
1. **Dark-First**: 深色模式不是反转，是从零设计的默认体验
2. **One-Handed**: 所有操作在拇指可达区域内完成
3. **Calm**: 无红色警告、无弹窗打断、无社交噪音

### 交互规范

| 交互 | 规范 | 原因 |
|------|------|------|
| 快速记录 | 1-tap开始，自动保存 | 凌晨3点单手操作 |
| 条目编辑 | 左滑→编辑/删除 | 符合iOS原生习惯 |
| 时间显示 | "14:30 (1h 23m ago)" | 具体时间+相对时间双格式 |
| 计时器 | Live Activity锁屏持续 | 离开APP不中断 |
| 夜间模式 | 永久深色，自动检测 | OLED省电+护眼 |
| 通知 | 仅Feed/Diaper超时提醒 | 不打扰，仅关键提醒 |
| 字体 | Rounded Design + Dynamic Type | 亲和力+可访问性 |
| 触控区域 | 最小44×44pt | Apple HWC标准 |
| 动画 | 轻柔弹簧动画(0.3s) | 不刺激，calm |
